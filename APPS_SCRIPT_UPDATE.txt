const ALERT_SUBJECT = 'Formosan Termite Swarm Alert \u2014 Kill Your Porch Lights Tonight';
const ALERT_BODY =
  'Swarm conditions are active in Southeast Texas tonight.\n\n' +
  'Warm, humid air and dropping pressure mean Formosan subterranean termites are likely swarming at dusk. ' +
  'They\u2019re drawn to any exterior light \u2014 porch lights, garage lights, landscape fixtures, even interior lights visible through windows.\n\n' +

function doPost(e) {
  const data = e.parameter;

  // --- AUTO-SEND: triggered by admin page with secret + send=1 ---
  if (data.secret === SECRET && data.send === '1') {
    // If specific emails are passed, send only to those; otherwise send to all.
    let emails;
    if (data.emails && data.emails.trim()) {
      emails = data.emails.split(',').map(e => e.trim()).filter(e => e.includes('@'));
    } else {
      const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
      const rows = sheet.getDataRange().getValues().slice(1);
      emails = rows.map(r => r[1]).filter(addr => addr && addr.toString().includes('@'));
    }

    let sent = 0;
    for (const addr of emails) {
      try {
        GmailApp.sendEmail(addr, ALERT_SUBJECT, ALERT_BODY);
        sent++;
      } catch (err) {
        // skip invalid addresses, keep going
      }
    }
    return ContentService.createTextOutput(
      JSON.stringify({ result: 'sent', count: sent })
    ).setMimeType(ContentService.MimeType.JSON);
  }

  // --- SIGNUP: regular form submission from the website ---
  const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
  sheet.appendRow([
    new Date(),
    data.email || '',
    data.name || '',
    data.area || '',
    (data.alerts || '')
  ]);
  return ContentService.createTextOutput(
    JSON.stringify({ result: 'success' })
  ).setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  // Serves subscriber data to the admin page (protected by secret)
  if (!e || !e.parameter || e.parameter.secret !== SECRET) {
    return ContentService.createTextOutput(
      JSON.stringify({ error: 'unauthorized' })
    ).setMimeType(ContentService.MimeType.JSON);
  }
  const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
  const allData = sheet.getDataRange().getValues();
  const rows = allData.slice(1);

  const emails = rows
    .map(r => r[1])
    .filter(addr => addr && addr.toString().includes('@'));

  const result = { emails: emails };
  if (e.parameter.full === '1') {
    result.rows = rows;
  }

  return ContentService.createTextOutput(
    JSON.stringify(result)
  ).setMimeType(ContentService.MimeType.JSON);
}
