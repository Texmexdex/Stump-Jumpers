// Run THIS once from the editor to authorize Gmail + Sheets access.

function setup() {
  const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
  // Ensure a header row exists
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['Date', 'Email', 'Name', 'Area', 'Alerts']);
  }
  // Send yourself a test so Gmail permission is granted
  GmailApp.sendEmail(Session.getActiveUser().getEmail(), 'Swarm Alert setup OK', 'Backend is connected and Gmail sending works.');
}

function doPost(e) {
  // Website signup form posts here -> append a row
  const data = (e && e.parameter) ? e.parameter : {};
  const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
  if (sheet.getLastRow() === 0) {
    sheet.appendRow(['Date', 'Email', 'Name', 'Area', 'Alerts']);
  }
  sheet.appendRow([
    new Date(),
    data.email || '',
    data.name || '',
    data.area || '',
    (data.alerts || '')
  ]);
  return ContentService.createTextOutput(
    JSON.stringify({ result: 'success' })
  ).setMimeType(ContentService.MimeType.JSON);
}

function doGet(e) {
  const p = (e && e.parameter) ? e.parameter : {};
  if (p.secret !== SECRET) {
    return json({ error: 'unauthorized' });
  }

  const sheet = SpreadsheetApp.openById(SHEET_ID).getActiveSheet();
  const rows = sheet.getDataRange().getValues().slice(1); // skip header row

  // ----- SEND -----
  if (p.send === '1') {
    let emails = [];
    if (p.rows && p.rows.toString().trim()) {
      const idxs = p.rows.toString().split(',').map(function (n) { return parseInt(n, 10); });
      idxs.forEach(function (idx) {
        if (idx >= 0 && idx < rows.length) {
          const addr = rows[idx][1];
          if (addr && addr.toString().indexOf('@') > -1) emails.push(addr.toString().trim());
        }
      });
    } else {
      emails = rows.map(function (r) { return r[1]; })
                   .filter(function (a) { return a && a.toString().indexOf('@') > -1; })
                   .map(function (a) { return a.toString().trim(); });
    }
    let sent = 0;
    emails.forEach(function (addr) {
      try { GmailApp.sendEmail(addr, ALERT_SUBJECT, ALERT_BODY); sent++; } catch (err) {}
    });
    return json({ result: 'sent', count: sent });
  }

  // ----- VIEW LIST -----
  const emailList = rows.map(function (r) { return r[1]; })
                        .filter(function (a) { return a && a.toString().indexOf('@') > -1; });
  const result = { emails: emailList };
  if (p.full === '1') result.rows = rows;
  return json(result);
}

function json(obj) {
  return ContentService.createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
